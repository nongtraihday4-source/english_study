🔍 Quyết định kiến trúc (Strict Rationale)
Phương án

Giải pháp thay thế
Per-question API call
	
Latency cao, phá vỡ UX, tăng tải server tuyến tính
	
Pool-based adaptive: Backend trả pool nhóm theo difficulty. Frontend điều hướng local. Backend vẫn là source of truth khi submit.
IRT/ML calibration
	
Cần >10k attempts, calibration phức tạp, overkill giai đoạn này

🔹 Bước 1: Cập nhật Quiz State & Adaptive Logic (GrammarDetailView.vue)
Thay thế toàn bộ khối Quiz State và các hàm liên quan trong <script setup> bằng đoạn sau:
// ── Adaptive Quiz State ──────────────────────────────────────────────────────
const quizPool = ref({ 1: [], 2: [], 3: [] }) // Grouped by difficulty
const currentDifficulty = ref(2) // Start Medium
const streak = ref(0)
const adaptiveHistory = ref([]) // source_id đã hỏi
const currentQuestion = ref(null)
const quizQuestions = ref([]) // Giữ lại để submit payload
const quizSaving = ref(false)
const quizSaved = ref(false)
const quizResult = ref(null)
const remedialRules = ref({})

const quizDone = computed(() => 
  quizQuestions.value.length >= 8 || !currentQuestion.value
)

const quizScoreDisplay = computed(() =>
  quizQuestions.value.filter(q => q.is_correct).length
)
const quizPercentDisplay = computed(() =>
  quizQuestions.value.length ? Math.round((quizScoreDisplay.value / quizQuestions.value.length) * 100) : 0
)

const hasActiveRemedial = computed(() => 
  Object.values(remedialRules.value).some(r => r.locked)
)

// ── Adaptive Routing ─────────────────────────────────────────────────────────
function pickNextQuestion() {
  let pool = quizPool.value[currentDifficulty.value]
  if (!pool || pool.length === 0) {
    // Fallback: nearest difficulty
    const fallback = currentDifficulty.value === 1 ? 2 : currentDifficulty.value - 1
    pool = quizPool.value[fallback] || []
  }
  const available = pool.filter(q => !adaptiveHistory.value.includes(q.source_id))
  if (available.length === 0) return null
  return available[Math.floor(Math.random() * available.length)]
}

function adjustDifficulty(isCorrect) {
  streak.value = isCorrect ? streak.value + 1 : streak.value - 1
  
  if (streak.value >= 3) {
    currentDifficulty.value = Math.min(3, currentDifficulty.value + 1)
    streak.value = 0
  } else if (streak.value <= -2) {
    currentDifficulty.value = Math.max(1, currentDifficulty.value - 1)
    streak.value = 0
  }
}

function selectAnswer(oi) {
  if (!currentQuestion.value || currentQuestion.value.selected !== null) return
  
  const q = currentQuestion.value
  q.selected = oi
  const isCorrect = oi === q.correct_index
  q.is_correct = isCorrect
  
  adjustDifficulty(isCorrect)
  adaptiveHistory.value.push(q.source_id)
  quizQuestions.value.push(q)
  
  // Delay next question for UX feedback
  setTimeout(() => {
    currentQuestion.value = pickNextQuestion()
  }, 800)
}

function optionClass(q, oi) {
  if (q.selected === null) return 'hover:bg-white/5 border border-transparent'
  if (oi === q.correct_index) return 'correct-opt'
  if (oi === q.selected) return 'wrong-opt'
  return 'opacity-50'
}

function resetQuiz() {
  quizPool.value = { 1: [], 2: [], 3: [] }
  currentDifficulty.value = 2
  streak.value = 0
  adaptiveHistory.value = []
  quizQuestions.value = []
  currentQuestion.value = null
  quizSaved.value = false
  quizResult.value = null
  remedialRules.value = {}
  fetchQuiz(topic.value?.slug)
}

// ── Backend Integration ──────────────────────────────────────────────────────
async function fetchQuiz(slug) {
  fetchingQuiz.value = true
  try {
    const res = await grammarApi.getQuizQuestions(slug)
    const data = res.data?.data ?? res.data ?? []
    
    // Group by difficulty
    quizPool.value = { 1: [], 2: [], 3: [] }
    data.forEach(q => {
      const diff = q.difficulty || 2
      quizPool.value[diff].push(reactive({ ...q, selected: null, is_correct: null }))
    })
    
    currentQuestion.value = pickNextQuestion()
  } catch (e) {
    console.error('Failed to fetch quiz:', e)
    quizPool.value = { 1: [], 2: [], 3: [] }
    currentQuestion.value = null
  } finally {
    fetchingQuiz.value = false
  }
}

async function submitQuizToBackend() {
  quizSaving.value = true
  try {
    const answers = quizQuestions.value.map(q => ({
      question_source_id: q.source_id,
      selected_option: q.options[q.selected]
    }))
    const idempotencyKey = `quiz-${topic.value.slug}-${Date.now()}`
    const res = await grammarApi.submitQuiz(topic.value.slug, {
      idempotency_key: idempotencyKey,
      answers: answers
    })
    const data = res.data?.data ?? res.data
    quizResult.value = data
    quizSaved.value = true
    remedialRules.value = data.remedial_rules || {}
  } catch (e) {
    console.error('Submit quiz failed:', e)
  } finally {
    quizSaving.value = false
  }
}

// Auto-submit when adaptive flow ends
watch(quizDone, async (done) => {
  if (!done || !isLoggedIn.value || !topic.value || quizSaved.value) return
  await submitQuizToBackend()
})

🔍 LƯU Ý KỸ THUẬT & RỦI RO
Quyết định
	
Lý do
	
Rủi ro & Mitigation
Pool-based + Client routing
	
Tránh latency per-question API. UX mượt, backend vẫn validate khi submit.
	
User inspect network thấy pool → Không ảnh hưởng pedagogy. Điểm số chỉ tính khi submit & backend chấm.
Streak threshold +3 / -2
	
Tăng độ khó chậm hơn giảm độ khó. Tránh frustrate khi user gặp câu khó bất ngờ.
	
Có thể tune thành +4/-2 nếu hệ thống báo "tăng độ khó quá nhanh".
Fallback nearest difficulty
	
Không bao giờ block user khi pool cạn. Giữ flow liên tục.
	
Có thể lặp lại câu đã hỏi nếu pool quá nhỏ → adaptiveHistory ngăn repeat. Phase 3.4 sẽ mở rộng pool generation.
Step-by-step rendering
	
Bắt buộc cho adaptive. Hiển thị tất cả cùng lúc phá vỡ logic điều chỉnh độ khó.
	
UX thay đổi so với cũ → User quen nhanh. Feedback immediate + streak bar giữ engagement.
